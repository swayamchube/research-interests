\begin{definition}[Multivariate Cyclotomic Polynomial]
    Let $\Phi_n\in\Z[x]$, define $\Phi_n\in\Z[x,y]$ as 
    \begin{equation*}
        \Phi_n(x, y) = y^{\varphi(n)}\Phi_n\left(\frac{x}{y}\right)
    \end{equation*}
\end{definition}
